      PPAマクロでこんな感じです
      var
      str_fn:String;
      begin
      str_fn := S_GetFilename();
      S_FileSave();
      S_FileCloseOpen();
      S_SelectAll(0);
      S_Copy(0);
      S_FileClose();
      S_FileOpen( str_fn );
      S_Paste(0);
      end;
