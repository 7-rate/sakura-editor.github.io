      function AutoCompIndent(ch)
      {
      var ch = Indent.GetChar();
      if( ch == "\r" ){
      return;
      }
      var x = Editor.ExpandParameter("$x") - 0;
      var s = Editor.GetLineStr(0).substr(0, x);
      var re = s.match(/[\w]{0,100}$/)
      if( re && 3 < re[0].length ){
      Editor.Complete();
      }
      }
