    <tr><td>2-1</td><td>2-2</td></tr>

    <tr><td>1-1</td></tr><tr><td>2-1</td></tr><tr><td>3-1</td></tr>

    <tr>(?=(.(?!<tr>))*2-1).*?></tr>
    <tr>(.(?!<tr>))*2-1.*?</tr>
