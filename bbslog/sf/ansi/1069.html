      /*! &の二重化

      @param org [in] 変換したい文字列
      @param out [out] 返還後の文字列を入れるバッファ

      @note 変換後のデータは最大で元の文字列の2倍になる

      @author genta
      @date 2002/01/04
      */
      void dupamp(const char* org, char* out)
      {
      for( ; *org != '\0'; ++org, ++out ){
      *out = *org;
      if( *org == '&' ){
      *(++out) = '&';
      }
      }
      *out = '\0';
      }
