  • [3857] サクラエディタでGrep ごく 2005年02月22日 11:27

    昔どこかの掲示板に書き込みしたことがあると思いますが、始めましてごくです。
    私はマクロの弱さからサクラエディタの使用をやめてしまったのですが、HDDから秀丸でGrepのパクリが出てきたのでこちらに貼り付けさせていただきます。
    すでに同等の機能が実装されていたり、起動オプションが変更されていたらごめんなさい。
    GrepHere.infとして保存し右クリックでインストールでOKなはずです。

    ;
    ; GrepHere.inf - Explorerから秀丸でGrep Version 0.03
    ; をパクってGrepHere.inf - ExplorerからSakura EditorでGrep Version 0.01
    ; Copyright(C) 1998 ymori@nisiq.net
    ; Copyright(C) 2002 gok

    [version]
    signature="$CHICAGO$"

    [DefaultInstall]
    CopyFiles=Inf.Copy
    AddReg=GrepHere.Reg

    [DefaultInstall.ntx86]
    CopyFiles=Inf.Copy
    AddReg=GrepHere.Reg.ntx86

    [DefaultUninstall]
    DelFiles=Inf.Copy
    DelReg=GrepHere.Reg

    [DefaultUninstall.ntx86]
    DelFiles=Inf.Copy
    DelReg=GrepHere.Reg.ntx86

    [Inf.Copy]
    GREPHERE.INF
    ; GREPHERE.TXT

    [SourceDisksNames]
    55=%TITLE%,"",1

    [SourceDisksFiles]
    Inf.Copy=55

    [DestinationDirs]
    Inf.Copy=17

    [GrepHere.Reg]
    HKCR,%DIRSG%
    HKCR,%DIRSG%,,,%MENU%
    HKCR,%DIRSG%\command\,,,%CMD%
    HKCR,%DRVSG%
    HKCR,%DRVSG%,,,%MENU%
    HKCR,%DRVSG%\command\,,,%CMD%
    HKLM,%SMWCUG%
    HKLM,%SMWCUG%,DisplayName,,%TITLE%
    HKLM,%SMWCUG%,UninstallString,,"rundll.exe setupx.dll,InstallHinfSection DefaultUninstall 132 %17%\GREPHERE.INF"

    [GrepHere.Reg.ntx86]
    HKCR,%DIRSG%
    HKCR,%DIRSG%,,,%MENU%
    HKCR,%DIRSG%\command\,,,%CMD%
    HKCR,%DRVSG%
    HKCR,%DRVSG%,,,%MENU%
    HKCR,%DRVSG%\command\,,,%CMD%
    HKLM,%SMWCUG%
    HKLM,%SMWCUG%,DisplayName,,%TITLE%
    HKLM,%SMWCUG%,UninstallString,,"RunDll32.exe syssetup.dll,SetupInfObjectInstallAction DefaultUninstall.ntx86 132 %17%\GREPHERE.INF"

    [Strings]
    DIRSG="DIRECTORY\shell\GrepHere"
    DRVSG="DRIVE\shell\GrepHere"
    MENU="Grep(&G)"
    CMD="SAKURA.EXE -GREPDLG -GREPMODE -GFILE=""*.*"" -GFOLDER=""%1"" -GOPT=S"
    SMWCUG="Software\Microsoft\Windows\CurrentVers
    ion\Uninstall\GrepHere"
    TITLE="サクラエディタでGrep"
