    // タイトル：固定長テキストデータを改行
    // LEFT_DOWN.PPA
    // マクロを実行前に、カーソルを
    // 最初に改行させたい位置に置きます。
    // あらかじめ途中に改行の有るデータはそこまでで終了します。
    var s1, s2 : String;
    begin
    while True do
    begin
    S_Char(13);
    S_Left();
    s1 := S_ExpandParameter('$x');
    S_Down();
    s2 := S_ExpandParameter('$x');
    if StrToInt(s1) <> StrToInt(s2) then
    break; // EOFか、または全角文字で終わる。
    end;
    end.
