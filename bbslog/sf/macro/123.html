    var len = inp.length;
    var i;

    if ( len == 1 && inp >= "A" ) {
    i = inp.charCodeAt(0) + 1;
    inp = String.fromCharCode(i);
    }
    else {
    inp = String(Number(inp)+1);
    }
    Editor.InsText(inp); // 変更
    for ( i = 0; i < len; i++ ) { // 変更部分を選択
