    ' 編集中のファイルのみを対象に選択文字列でGrep実行（ただし保存時点）
    sKey = inputbox("","Grep文字列の指定", ExpandParameter(Cstr("$C")))
    sFile = ExpandParameter(Cstr("$f"))
    sPath = ExpandParameter(Cstr("$F"))

    if len(sKey) > 0 and sFile <> sPath then
    msgbox spath & "," & sfile
    sDir = Left(CStr(sPath), Len(sPath) - Len(sFile) - 1)

    '01100011
    '0111000=56
    '0011000=24
    Grep CStr(sKey), CStr(sFile), CStr(sDir), CLng(56)
    end if
